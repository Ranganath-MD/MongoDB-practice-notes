Collections
MongoDB stores documents in collections. Collections are analogous to tables in relational databases.


db.createCollection("marvel")
db.createUser()
db.createView()
db.createRole()

create collection
db.createCollection("marvel")

drop collection
db.getCollection("marvel").drop()
or////////////////////////////////
db.marvel.drop()
